% More stuff for the geometry of the page (e.g. 1.5 line spacing).
\usepackage[a4paper, portrait, margin=1in]{geometry}
\renewcommand{\baselinestretch}{1.5}
\usepackage{newtxtext,newtxmath}

% Language and input details.
\usepackage[catalan]{babel}
\usepackage[utf8]{inputenc}

% Do not do funny things with links.
\usepackage{xcolor}
\hypersetup{colorlinks=false, pdfborderstyle={/S/U/W 1}, urlbordercolor=gray, pdfborder=0 0 1}
\hypersetup{hidelinks}

% Fancy headers and footers.
\usepackage{fancyhdr}

% Redefine the plain page style with fancy header in it.
\fancypagestyle{plain}{%
  \fancyhf{}%
  \fancyfoot[C]{\footnotesize —\hspace{2mm}\thepage\hspace{2mm}—}%
}

\renewcommand{\fancyshape}{\scshape}
\renewcommand{\textfancy}{\textsc}

% Used to tweak various settings around titles (e.g. the format of chapter titles).
\usepackage{titlesec}
\titleformat{\chapter}{\fancyshape\large}{\thechapter.}{20pt}{\large\fancyshape}
\titleformat{\section}{\fancyshape\normalsize}{\thechapter.\thesection}{16pt}{\normalsize\fancyshape}

% Used to tweak the style of the ToC.
\usepackage{tocloft}

% Bibliography
% \usepackage[natbibapa]{apacite}
\usepackage{natbib}
\bibliographystyle{newapa}

% Useful package to replicate text around (e.g. poem titles).
\usepackage{clipboard}

% The verse environment
\usepackage{verse}
\setlength{\stanzaskip}{0.75\baselineskip} % The distance between stanzas
\newcounter{poem}

% Include the epigraph environment.
\usepackage{epigraph}
\setlength\epigraphrule{0pt}
\setlength\epigraphwidth{.6\textwidth}
